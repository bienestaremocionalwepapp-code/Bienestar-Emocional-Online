function getDisponibilidad(soloActivos) { var sh = crearPestanaDisponibilidad(); var data = sh.getDataRange().getValues(); var hoy = new Date(); hoy.setHours(0,0,0,0); var result = []; for (var i = 1; i < data.length; i++) { var row = data[i]; if (!row[0]) continue; // Activo — acepta TRUE, true, 1 var activo = String(row[5]).toLowerCase().trim(); if (soloActivos && activo !== 'true' && activo !== '1') continue; // Fecha — viene como Date object de Sheets var fechaRaw = row[1]; if (!fechaRaw) continue; var fechaObj = new Date(fechaRaw); fechaObj.setHours(0,0,0,0); // Filtrar fechas pasadas if (soloActivos && fechaObj < hoy) continue; var fecha = Utilities.formatDate(fechaObj, Session.getScriptTimeZone(), 'yyyy-MM-dd'); result.push({ id: row[0], fecha: fecha, hora_inicio: row[2], hora_fin: row[3], nota: row[4], activo: 'true' }); } result.sort(function(a,b){ return (a.fecha + (a.hora_inicio||'')).localeCompare(b.fecha + (b.hora_inicio||'')); }); return result; }
